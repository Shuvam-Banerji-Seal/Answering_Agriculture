\begin{frame}{Technology Stack Overview}
\begin{columns}

\begin{frame}{Embedding Generation System}
\begin{columns>
\begin{column>{0.5\textwidth}
\textbf{System Architecture:}
\begin{itemize>
\item \textbf{Model:} Qwen3-Embedding-8B
\item \textbf{Dimension:} 4096-dimensional vectors
\item \textbf{Chunking:} Overlapping token-based (256 tokens, 25 overlap)
\item \textbf{Processing:} Mean pooling of hidden states
\item \textbf{Optimization:} GPU acceleration, batch processing
\end{itemize>

\textbf{Performance Metrics:}
\begin{itemize>
\item \textbf{Speed:} 50-100 records/minute (GPU)
\item \textbf{Memory:} ~8GB GPU for model
\item \textbf{Index Build:} 1M embeddings in 30 seconds
\item \textbf{Search:} <1ms per query (FAISS)
\end{itemize>
\end{column>
\begin{column>{0.5\textwidth}
\textbf{Data Pipeline:}
\begin{itemize>
\item JSONL dataset input
\item Intelligent text chunking
\item Agricultural metadata extraction
\item FAISS index generation
\item Quality assurance \& validation
\end{itemize>

\textbf{Output Structure:}
\begin{itemize>
\item \texttt{embeddings.npy} - Vector arrays
\item \texttt{faiss\_index.bin} - Search index
\item \texttt{metadata.json} - Human-readable
\item \texttt{metadata.pkl} - Binary format
\item \texttt{config.json} - Configuration
\item \texttt{summary\_stats.json} - Analytics
\end{itemize>
\end{column>
\end{columns>
\end{frame>
